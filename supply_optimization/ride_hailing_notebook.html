df[['start_lat', 'start_lng', 'end_lat', 'end_lng']] = df[['start_lat', 'start_lng', 'end_lat', 'end_lng']].round(4)

# Extract time and date-related features
df['pickup_time'] = pd.to_datetime(df['start_time']).dt.time

df['hourly_segments'] = df['hour'].map(lambda x: f"H{(x // 4) + 1}")

3.2 Outlier treatment

Reasons for Choosing Percentile-Based Capping

  • Handles Skewed Data: It effectively manages non-normal distributions, common in ride values.
  • Maintains Data Size: Capping preserves data points, preventing data loss.
  • Easy to Understand: The method is straightforward and easily communicated to stakeholders.
  • Customizable: Percentile thresholds can be adjusted based on specific business needs.

3.3 Location Clustering

Why Choose MiniBatchKMeans

MiniBatchKMeans is often chosen for its efficiency and speed when dealing with large datasets. It works on small random batches of the data, significantly reducing the computational burden compared to standard K-Means.

Its implementation also tends to require less fine-tuning of hyperparameters compared to other clustering methods, which can simplify the clustering process in practical applications.

# Model evaluation function
def model_creation_n_evaluation(algorithm, X_train, y_train, X_test, y_test):
    # Initialize and train the model
    model = algorithm
    model.fit(X_train, y_train)
    
    # Predict on both training and test datasets
    y_train_pred = model.predict(X_train)
    y_test_pred = model.predict(X_test)

    # Calculate metrics for training data
    r2_train = r2_score(y_train, y_train_pred)
    mse_train = mean_squared_error(y_train, y_train_pred)
    rmse_train = math.sqrt(mse_train)

    # Calculate metrics for test data
    r2_test = r2_score(y_test, y_test_pred)
    mse_test = mean_squared_error(y_test, y_test_pred)
    rmse_test = math.sqrt(mse_test)

    # Display results in a table
    table = PrettyTable()
    table.field_names = ["Metric", "Training Value", "Test Value"]
    table.add_row(["R", r2_train, r2_test])
    table.add_row(["MSE", mse_train, mse_test])
    table.add_row(["RMSE", rmse_train, rmse_test])

    algorithm_name = str(algorithm).split("(")[0]
    table.title = algorithm_name
    print(table)

def demand_data_prep(input_time):
    # Create DataFrame with repeated input_time and pickup_cluster range
    df = pd.DataFrame({
        "start_time": [input_time] * 30,
        "pickup_cluster": range(30)
    })

    # Convert 'start_time' to datetime and handle errors
    df['start_time'] = pd.to_datetime(df['start_time'], errors='coerce').dt.strftime('%Y-%m-%d %H:%M:%S')

    # Extract features from 'start_time'
    df['pickup_time'] = pd.to_datetime(df['start_time']).dt.time
    df['day_of_month'] = pd.to_datetime(df['start_time']).dt.day
    df['year'] = pd.to_datetime(df['start_time']).dt.year
    df['month'] = pd.to_datetime(df['start_time']).dt.month
    df['weekday'] = pd.to_datetime(df['start_time']).dt.weekday
    df['hour'] = pd.to_datetime(df['start_time']).dt.hour

    # Create quarterly and hourly segment mappings
    df['monthly_quarter'] = df['month'].map({1: 'Q1', 2: 'Q1', 3: 'Q1', 4: 'Q2', 5: 'Q2', 6: 'Q2',
                                              7: 'Q3', 8: 'Q3', 9: 'Q3', 10: 'Q4', 11: 'Q4', 12: 'Q4'})
    df['hourly_segments'] = df['hour'] // 4  # 0-3 -> H1, 4-7 -> H2, etc.
    df['hourly_segments'] = df['hourly_segments'].map(lambda x: f'H{x + 1}')

    hs = df['hourly_segments'][0]
    df[f'hourly_segments_{hs}'] = 1
 
    columns_to_keep = ['pickup_cluster', 'day_of_month', 'weekday', 'hour',
       'hourly_segments_H1', 'hourly_segments_H2', 'hourly_segments_H3',
       'hourly_segments_H4', 'hourly_segments_H5', 'hourly_segments_H6']

    for col in columns_to_keep:
        if col not in df.columns:
            df[col] = 0
    
    return df[columns_to_keep]

def create_demand_prediction(df, model):
    # Predict demand and create a DataFrame with the results
    df['demand_count'] = model.predict(df).astype(int)
    return df[['pickup_cluster', 'demand_count']]
    
def create_sample_supply_data(total_units):
    # Randomly distribute total_units across 30 pickup clusters
    supply_counts = np.random.multinomial(total_units, [1/30] * 30)
    return pd.DataFrame({'pickup_cluster': range(30), 'supply_count': supply_counts})

def ride_value(demand_table, df_model_agg):
    # Merge demand table with aggregated model values
    input_df_short = demand_table[["pickup_cluster", "day_of_month", "weekday", "hour"]]
    merged = pd.merge(input_df_short, df_model_agg, on=["pickup_cluster", "day_of_month", "weekday", "hour"], how="inner")
    merged["avg_ride_value"] = merged["value_sum_sum"] / merged["demand_count_sum"]
    return merged

def calculate_expected_ride_metrics(input_df, rider_value_data, supply_demand_table):
    # Calculate expected ride metrics and supply-demand ratios
    rider_value_table = pd.merge(input_df, rider_value_data,  on=["pickup_cluster","day_of_month","weekday","hour"], how="inner")[["pickup_cluster", "avg_ride_value", "p25_ride_value", "p75_ride_value"]]
    supply_demand_table_w_values = pd.merge(supply_demand_table, rider_value_table, on=["pickup_cluster"], how="inner")

    supply_demand_table_w_values["supply_demand_ratio"] = supply_demand_table_w_values["supply_count"] / supply_demand_table_w_values["demand_count"]
    supply_demand_table_w_values["supply_demand_diff"] = supply_demand_table_w_values["demand_count"] - supply_demand_table_w_values["supply_count"]

    # Set expected ride value based on supply-demand ratio thresholds
    supply_demand_table_w_values["expected_ride_value"] = np.select(
        [supply_demand_table_w_values["supply_demand_ratio"] < 0.25,
         supply_demand_table_w_values["supply_demand_ratio"] > 0.75],
        [supply_demand_table_w_values["p75_ride_value"], supply_demand_table_w_values["p25_ride_value"]],
        default=supply_demand_table_w_values["avg_ride_value"]
    )

    # Calculate expected wait time, ensuring it stays within bounds
    supply_demand_table_w_values["expected_wait_time"] = 60 / supply_demand_table_w_values["supply_demand_diff"]
    supply_demand_table_w_values["expected_wait_time"] = np.where( (supply_demand_table_w_values["expected_wait_time"] < 0) |
                                                                   (supply_demand_table_w_values["expected_wait_time"] > 60), 60,
                                                                    supply_demand_table_w_values["expected_wait_time"])
    
    supply_demand_table_w_values["expected_ride_value"] = supply_demand_table_w_values["expected_ride_value"].round(2)
    supply_demand_table_w_values["expected_wait_time"] =  np.ceil(supply_demand_table_w_values["expected_wait_time"]).astype(int)

    
    return supply_demand_table_w_values[["pickup_cluster", "demand_count", "supply_count", "expected_wait_time", "expected_ride_value"]]
    
def find_starting_cluster(input_coordinates, cluster_bounds):
    # Find the cluster for the given input coordinates
    lat, lon = map(float, input_coordinates)
    selected_cluster = cluster_bounds[
        (cluster_bounds['min_latitude'] <= lat) &
        (cluster_bounds['max_latitude'] >= lat) &
        (cluster_bounds['min_longitude'] <= lon) &
        (cluster_bounds['max_longitude'] >= lon)
    ]
    return selected_cluster['pickup_cluster'].iloc[0] if not selected_cluster.empty else None

def run(input_time, input_coordinates, total_units, rider_value_data, cluster_bounds, model):
    # Step 1: Prepare the demand data
    input_df = demand_data_prep(input_time)

    
    # Step 2: Predict demand based on input data
    demand_table = create_demand_prediction(input_df, model)
    

    # Step 3: Create supply data, distributing 511 units across clusters
    supply_table = create_sample_supply_data(total_units)
    
    # Step 4: Merge demand and supply tables on pickup_cluster
    supply_demand_table = pd.merge(demand_table, supply_table, on="pickup_cluster", how="inner")
    
    supply_demand_table_w_values = calculate_expected_ride_metrics(input_df,rider_value_data,supply_demand_table)
    #print(supply_demand_table_w_values)
    
    input_cluster = find_starting_cluster(input_coordinates, cluster_bounds)

    # Sorting the DataFrame based on expected wait time and expected ride value
    top_clusters = supply_demand_table_w_values.sort_values(by=['expected_wait_time', 'expected_ride_value'], ascending=[True, False])\
                                                .head(3).reset_index(drop=True)
    
    current_cluster = supply_demand_table_w_values[supply_demand_table_w_values["pickup_cluster"] == input_cluster].reset_index(drop=True)
    current_cluster.columns = ['current_' + col for col in current_cluster.columns]
    current_cluster = pd.concat([current_cluster] * 3, ignore_index=True)
    recommendation_table = pd.concat([top_clusters, current_cluster.reset_index(drop=True)], axis=1)

    recommendation_table["waiting_time_change"] = recommendation_table["current_expected_wait_time"] / recommendation_table["expected_wait_time"]
    recommendation_table["rider_value_change_perc"] = (1 - recommendation_table["current_expected_ride_value"] / recommendation_table["expected_ride_value"]) * 100
    recommendation_table = recommendation_table[["pickup_cluster","expected_wait_time","expected_ride_value","waiting_time_change","rider_value_change_perc"]]
    return recommendation_table
